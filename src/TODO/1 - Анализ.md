  🔴 Выявленные проблемы

  1. Архитектурные проблемы

  Критические:
  - ❌ Нарушение SOLID - SystemInfoRepository содержит бизнес-логику (форматирование)
  - ❌ Статические методы - невозможно тестировать, нет Dependency Injection
  - ❌ Пустой Service - SysInfoService инжектится но не используется
  - ❌ Контроллер → Repository - нарушение слоистой архитектуры (должен быть через Service)
  - ❌ Дублирование кода - данные собираются дважды (для view и для AJAX)
  - ❌ Жесткая привязка к Linux - читает /proc/* файлы напрямую

  Средние:
  - ⚠️ Нет обработки ошибок - если /proc/* недоступны, будут warnings
  - ⚠️ Нет кеширования - каждый запрос читает файловую систему
  - ⚠️ Смешение ответственности - форматирование данных в Repository

  2. UI/UX проблемы

  Критические:
  - ❌ Устаревший дизайн - старые HTML таблицы вместо Bootstrap 5 карточек
  - ❌ Самописный JavaScript - велосипед вместо стандартного jQuery
  - ❌ Плохая читаемость - нет группировки, все в одной куче
  - ❌ Не адаптивно - плохо выглядит на мобильных устройствах

  Средние:
  - ⚠️ Нет визуализации - нет графиков истории метрик
  - ⚠️ Нет алертов - не выделяются критические значения
  - ⚠️ Жесткий интервал - обновление каждую секунду (нельзя настроить)
  - ⚠️ Китайские комментарии в коде - мешает понимать

  3. Функциональные пробелы

  Критические (для Docker окружения):
  - ❌ Нет информации о Docker - контейнеры, их статусы, ресурсы
  - ❌ Нет информации о БД - MySQL, Redis статистика
  - ❌ Нет информации о Yii2 - кеш, сессии, очереди, миграции

  Средние:
  - ⚠️ Нет проверки здоровья - healthcheck сервисов
  - ⚠️ Нет логов - просмотр логов в реальном времени
  - ⚠️ Нет истории - невозможно увидеть метрики за период
  - ⚠️ Нет экспорта - невозможно выгрузить данные

  1. Архитектура (Clean Architecture)

  Новая структура:

  Besnovatyj/Info/
  ├── controllers/backend/
  │   └── SysInfoController.php          # Тонкий контроллер
  ├── services/
  │   ├── SysInfoService.php             # Бизнес-логика, координация

  │       ├── MetricProviderInterface.php

  │       ├── PHPMetricProvider.php       # PHP info
  │       ├── DockerMetricProvider.php    # Docker контейнеры
  │       ├── DatabaseMetricProvider.php  # MySQL, Redis
  │       ├── ApplicationMetricProvider.php # Yii2 специфика
  │       └── NetworkMetricProvider.php   # Сетевые интерфейсы
  ├── dto/                                # Data Transfer Objects
  │   ├── SystemInfoDto.php
  │   ├── MetricDto.php
  │   └── ...
  ├── repositories/
  │   ├── SystemInfoRepository.php        # Только чтение данных, без форматирования

      ├── MetricCardWidget.php
      ├── MetricChartWidget.php
      └── ...

  - ✅ Testable - можно мокать providers

 class SysInfo {
     constructor(config) {

             headers: {
                 'X-CSRF-Token': this.csrfToken,
                 'X-Requested-With': 'XMLHttpRequest',
             }
         });
         return response.json();
     }

     updateUI(metrics) {
         // Обновление всех метрик на странице
         this.updateProgressBars(metrics);
         this.chartManager.addDataPoint(metrics);
         this.alertManager.checkThresholds(metrics);
     }

         this.intervalId = setInterval(() => {
             this.fetchMetrics().then(data => this.updateUI(data));
         }, this.updateInterval);
     }

 Chart.js интеграция

 class ChartManager {
     constructor() {
         this.charts = new Map();
         this.maxDataPoints = 60; // 60 секунд истории
     }

     createChart(canvasId, label) {
         const ctx = document.getElementById(canvasId).getContext('2d');
         const chart = new Chart(ctx, {
             type: 'line',
             data: {
                 labels: [],
                 datasets: [{
                     label: label,
                     data: [],
                     borderColor: 'rgb(75, 192, 192)',
                     tension: 0.1
                 }]
             },
             options: {
                 responsive: true,
                 scales: {
                     y: { beginAtZero: true, max: 100 }
                 }
             }
         });

         this.charts.set(canvasId, chart);
     }

     addDataPoint(metrics) {
         this.charts.forEach((chart, id) => {
             const time = new Date().toLocaleTimeString();
             const value = metrics[id]; // Значение метрики

             chart.data.labels.push(time);
             chart.data.datasets[0].data.push(value);

             // Ограничение истории
             if (chart.data.labels.length > this.maxDataPoints) {
                 chart.data.labels.shift();
                 chart.data.datasets[0].data.shift();
             }

             chart.update('none'); // Без анимации для производительности
         });
     }
 }

 Система алертов

 class AlertManager {
     constructor() {
         this.thresholds = {
             cpu: {warning: 70, critical: 85},
             memory: {warning: 75, critical: 90},
             disk: {warning: 80, critical: 95},
         };
         this.notifiedMetrics = new Set();
     }

     checkThresholds(metrics) {
         Object.entries(metrics).forEach(([key, value]) => {
             if (this.thresholds[key]) {
                 const threshold = this.thresholds[key];

                 if (value >= threshold.critical) {
                     this.showAlert(key, value, 'critical');
                 } else if (value >= threshold.warning) {
                     this.showAlert(key, value, 'warning');
                 } else {
                     this.notifiedMetrics.delete(key);
                 }
             }
         });
     }

     showAlert(metric, value, level) {
         const alertKey = `${metric}-${level}`;
         if (this.notifiedMetrics.has(alertKey)) return;

         this.notifiedMetrics.add(alertKey);

         showAlert({
             message: `${metric.toUpperCase()}: ${value}% (${level})`,
             type: level === 'critical' ? 'error' : 'warning',
             duration: 5000
         });
     }
 }

 class DatabaseMetricProvider implements MetricProviderInterface {
     private Connection $db;
     private Redis $redis;

     public function getMetrics(): array {
         return [
             'mysql' => $this->getMysqlStats(),
             'redis' => $this->getRedisStats(),
         ];
     }

     private function getMysqlStats(): array {
         $db = Yii::$app->db;

         return [
             'version' => $db->createCommand('SELECT VERSION()')->queryScalar(),
             'connections' => $db->createCommand('SHOW STATUS LIKE "Threads_connected"')->queryOne(),
             'database_size' => $this->getDatabaseSize(),
             'table_count' => $this->getTableCount(),
         ];
     }

     private function getRedisStats(): array {

             'keys' => $redis->executeCommand('DBSIZE'),
             'hit_rate' => $this->calculateHitRate($info),
         ];
     }
